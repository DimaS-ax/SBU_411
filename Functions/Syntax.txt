type name(parameters)
{
	.....;
	group-of-statements;
	.....;
}

type - тип возвращаемого значения
void - пустота
return - прерывает работу функции, возвращает значение на место вызова,
		 и возвращает управление на место вызова.

		 return value;
		 return;

name - имя функции;

parameters - список принимаемых параметров функции.
